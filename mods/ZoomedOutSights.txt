#<Zoomed Out Sights>

    #<Description>

        Seriously, how do holograms zoom in that far?

        Reduces the zoom of non-scope sights, similarly to other games where red dot sights don't provide additional zoom

        Made by TheDerpOfGames

    #</Description>

    #<Pistol Sights>

        #<Dahl Mini Red Dot>

            set GD_Weap_Pistol.Sight.Pistol_Sight_Dahl AttributeSlotUpgrades (SlotName="WeaponZoomFOV",GradeIncrease=4,bActivateSlot=True)

        #</Dahl Mini Red Dot>

        #<Jakobs Double Lense Sight>

            set GD_Weap_Pistol.Sight.Pistol_Sight_Jakobs AttributeSlotUpgrades (SlotName="WeaponZoomFOV",GradeIncrease=4,bActivateSlot=True)

        #</Jakobs Double Lense Sight>

        #<Maliwan Hologram Sight>

            set GD_Weap_Pistol.Sight.Pistol_Sight_Maliwan AttributeSlotUpgrades (SlotName="WeaponZoomFOV",GradeIncrease=4,bActivateSlot=True)

        #</Maliwan Hologram Sight>
